 820        private static VisualElement PerformPick(VisualElement root, Vector2 point, List<VisualElement> picked = null)
 821        {
 822            // Skip picking for elements with display: none
 823            if (root.resolvedStyle.display == DisplayStyle.None)
 824                return null;
 825
 826            if (root.pickingMode == PickingMode.Ignore && root.hierarchy.childCount == 0)
 827            {
 828                return null;
 829            }
 830
 831            if (!root.worldBoundingBox.Contains(point))
 832            {
 833                return null;
 834            }
 835
 836            // Problem here: everytime we pick, we need to do that expensive transformation.
 837            // The default Contains() compares with rect, while we could cache the rect in world space (transform 2 poin
 838            // here we have to transform 1 point at every call.
 839            // Now since this is a virtual, we can't just start to call it with global pos... we could break client code
 840            // EdgeControl and port connectors in GraphView overload this.
 841            Vector2 localPoint = root.WorldToLocal(point);
 842
 843            bool containsPoint = root.ContainsPoint(localPoint);
 844            // we only skip children in the case we visually clip them
 845            if (!containsPoint && root.ShouldClip())
 846            {
 847                return null;
 848            }
 849
 850            VisualElement returnedChild = null;
 851            // Depth first in reverse order, do children
 852            var cCount = root.hierarchy.childCount;
 853            for (int i = cCount - 1; i >= 0; i--)
 854            {
 855                var child = root.hierarchy[i];
 856                var result = PerformPick(child, point, picked);
 857                if (returnedChild == null && result != null && result.visible)
 858                {
 859                    if (picked == null)
 860                    {
 861                        return result;
 862                    }
 863
 864                    returnedChild = result;
 865                }
 866            }
 867
 868            if (root.enabledInHierarchy && root.visible && root.pickingMode == PickingMode.Position && containsPoint)
 869            {
 870                picked?.Add(root);
 871                if (returnedChild == null)
 872                    returnedChild = root;
 873            }
 874
 875            return returnedChild;
 876        }
