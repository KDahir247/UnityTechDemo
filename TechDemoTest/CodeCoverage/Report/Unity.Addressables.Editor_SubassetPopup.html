 954        /// <typeparam name="T">The type of the object to extract.</typeparam>

 960        {

 964                    return default(T);

 966                if (serializedObject == null)

 974                {

 978                    {

 981                        if (currName.EndsWith("]"))
 982                        {

 984                            if (arraySlice.Length >= 2)
 985                            {
